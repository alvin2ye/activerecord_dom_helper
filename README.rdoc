== Example

>> Book.first.dom_id
=> "book-1"
>> Book.new.dom_id
=> "book-new"
>> Book.new.dom_id('xx')
=> "xx-book-new"
>> Book.new.dom_class('xx')
=> "xx-book"
>> Book.new.dom_class
=> "book"
